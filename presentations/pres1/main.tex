\begin{frame}{Definicion del Modelo}
  \textbf{Conjuntos (Indices)}
  \begin{itemize}
    \item $I$: insumos o ingredientes
    \item $T$: dias de la semana, $t \in \{1,2,3,4,5,6,7\}$
    \item $M$: platillos o menus disponibles
    \item $K$: dias de vida util
  \end{itemize}

  \textbf{Parametros}
  \begin{itemize}
    \item $C_i$: costo de comprar una unidad del insumo $i$
    \item $\text{Receta}_{i,m}$: cantidad del insumo $i$ por porcion del platillo $m$
    \item $Demanda_t$: personas a alimentar el dia $t$
    \item $Donaciones_{i,t}$: insumo $i$ recibido gratis el dia $t$
    \item $Capacidad_i$: espacio maximo de bodega para el insumo $i$
    \item $Lote_i$: tamano del paquete del insumo $i$
    \item $\tau_i$: vida util maxima del insumo $i$
  \end{itemize}

\end{frame}

\begin{frame}{Definicion del Modelo}
  \textbf{Variables de Decision}
  \begin{itemize}
    \item $Z_{m,t} \ge 0$ (entera): porciones del platillo $m$ a cocinar el dia $t$
    \item $Y_{i,t} \ge 0$ (entera): paquetes a comprar del insumo $i$ el dia $t$
    \item $X_{i,t} \ge 0$ (continua): cantidad total comprada del insumo $i$ el dia $t$
    \item $Consumo_{i,t} \ge 0$ (continua): cantidad del insumo $i$ usada el dia $t$
    \item $Consumo\_Edad_{i,t,k} \ge 0$ (continua): consumo del insumo $i$ con $k$ dias de vida
    \item $Inv_{i,t,k} \ge 0$ (continua): inventario del insumo $i$ al final del dia $t$ con $k$ dias de vida
  \end{itemize}
\end{frame}


\begin{frame}{Funcion Objetivo y Restricciones}
  \textbf{Funcion Objetivo}
  \[
    \min Z = \sum_{t \in T} \sum_{i \in I} C_i \, X_{i,t}
  \]

  \textbf{Restricciones}
  \[
    \sum_{m \in M} Z_{m,t} \ge Demanda_t \quad \forall t \in T
  \]
  \[
    Consumo_{i,t} = \sum_{m \in M} \text{Receta}_{i,m} \, Z_{m,t} \quad \forall i \in I, \forall t \in T
  \]
  \[
    X_{i,t} = Lote_i \, Y_{i,t} \quad \forall i \in I, \forall t \in T
  \]
  \[
    \sum_{k \in K} Inv_{i,t,k} \le Capacidad_i \quad \forall i \in I, \forall t \in T
  \]
  \[
    Inv_{i,t,\tau_i} = X_{i,t} + Donaciones_{i,t} \quad \forall i \in I, \forall t \in T
  \]
  \[
    Inv_{i,t,k-1} = Inv_{i,t-1,k} - Consumo\_Edad_{i,t,k} \quad \forall i \in I, \forall t > 1, \forall k \in K
  \]
\end{frame}
